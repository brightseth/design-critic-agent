Session Workflow Example
// 1. Create session with specific goals
const session = await createSession({
  title: "Emerging Artists Showcase",
  goal: "Find 20 breakthrough works",
  targetCount: 20,
  criteria: {
    minQuality: 75,
    themes: ["experimental", "innovative"]
  }
})

// 2. Review works systematically
for (const workId of session.workQueue) {
  const work = await getWork(workId)
  const analysis = await analyzeWork(workId, "curation")
  
  // 3. Make decision
  const decision = evaluateWork(work, analysis)
  await recordDecision(session.id, workId, decision)
  
  // 4. Check progress
  if (session.accepted.length >= session.targetCount) {
    break
  }
}

Integration Guide

Complete examples for integrating the Curation API into your application.

Complete Nina Integration Client
class CurationClient {
  constructor(baseUrl = 'https://eden-genesis-registry.vercel.app/api/v1') {
    this.baseUrl = baseUrl
  }

  // Agent Works Access
  async getAgentWorks(agentId, filters = {}) {
    const params = new URLSearchParams(filters)
    return fetch(`${this.baseUrl}/agents/${agentId}/works?${params}`)
      .then(r => r.json())
  }

  // Analysis
  async analyzeWorks(agentId, workIds, type = 'registry') {
    return fetch(`${this.baseUrl}/agents/${agentId}/analyze`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ workIds, analysisType: type })
    }).then(r => r.json())
  }

  // Session Management
  async createSession(sessionData) {
    return fetch(`${this.baseUrl}/curation/sessions`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(sessionData)
    }).then(r => r.json())
  }

  async recordDecision(sessionId, decision) {
    return fetch(`${this.baseUrl}/curation/sessions/${sessionId}/decisions`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(decision)
    }).then(r => r.json())
  }

  // Collections
  async createCollection(collectionData) {
    return fetch(`${this.baseUrl}/collections`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(collectionData)
    }).then(r => r.json())
  }

  async addToCollection(collectionId, workData) {
    return fetch(`${this.baseUrl}/collections/${collectionId}/works`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(workData)
    }).then(r => r.json())
  }

  // Collaboration
  async createCollaboration(collaborationData) {
    return fetch(`${this.baseUrl}/collaborations`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(collaborationData)
    }).then(r => r.json())
  }

  async submitVote(collaborationId, vote) {
    return fetch(`${this.baseUrl}/collaborations/${collaborationId}/vote`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(vote)
    }).then(r => r.json())
  }

  // Analytics
  async getCuratorAnalytics(curatorId, period = 'all') {
    return fetch(`${this.baseUrl}/curation/analytics?curatorId=${curatorId}&period=${period}`)
      .then(r => r.json())
  }
}

// Usage Example
const curator = new CurationClient()

// Start curation workflow
async function curateForParisPhoto() {
  // 1. Create session
  const session = await curator.createSession({
    title: "Paris Photo 2025",
    goal: "Select 15 works for main gallery",
    targetCount: 15,
    criteria: {
      themes: ["identity", "transformation"],
      minQuality: 80
    }
  })

  // 2. Analyze candidates
  const analysis = await curator.analyzeWorks(
    'solienne-001',
    session.workQueue.slice(0, 50),
    'curation'
  )

  // 3. Review and decide
  for (const result of analysis.results) {
    if (result.analysis.exhibitionFit > 85) {
      await curator.recordDecision(session.id, {
        workId: result.workId,
        decision: 'accept',
        curatorId: 'nina'
      })
    }
  }

  // 4. Create exhibition collection
  const collection = await curator.createCollection({
    title: "Paris Photo 2025 - Main Gallery",
    type: "exhibition",
    status: "active"
  })

  // 5. Add accepted works
  for (const workId of session.accepted) {
    await curator.addToCollection(collection.id, {
      workId,
      curatorId: 'nina'
    })
  }

  return { session, collection }
}
Python Integration Example
import requests
import json
from typing import Dict, List, Optional

class GenesisRegistryCurator:
    def __init__(self, base_url: str = "https://eden-genesis-registry.vercel.app/api/v1"):
        self.base_url = base_url
        self.session = requests.Session()
    
    def get_agent_works(self, agent_id: str, **filters) -> Dict:
        """Fetch works from a specific agent"""
        response = self.session.get(
            f"{self.base_url}/agents/{agent_id}/works",
            params=filters
        )
        return response.json()
    
    def analyze_works(self, agent_id: str, work_ids: List[str], 
                     analysis_type: str = "registry") -> Dict:
        """Analyze works for curation"""
        response = self.session.post(
            f"{self.base_url}/agents/{agent_id}/analyze",
            json={
                "workIds": work_ids,
                "analysisType": analysis_type
            }
        )
        return response.json()
    
    def create_curation_session(self, **session_data) -> Dict:
        """Create a new curation session"""
        response = self.session.post(
            f"{self.base_url}/curation/sessions",
            json=session_data
        )
        return response.json()
    
    def record_decision(self, session_id: str, work_id: str, 
                       decision: str, curator_id: str) -> Dict:
        """Record a curation decision"""
        response = self.session.post(
            f"{self.base_url}/curation/sessions/{session_id}/decisions",
            json={
                "workId": work_id,
                "decision": decision,
                "curatorId": curator_id
            }
        )
        return response.json()

# Create a curation session

    if result["analysis"]["exhibitionFit"] > 80:

            result["workId"],

            "nina"
